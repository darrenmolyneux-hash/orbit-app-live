SELECT 
    c.collectionid,
    c.collection_ref,
    c.customerid,
    cust.customername,
    c.siteid,
    c.address1,
    c.address2,
    c.city,
    c.postcode,
    c.collectiondate,
    c.jobtype,
    c.createdby,
    c.collectiondescription,
    c.notes,
    c.status
FROM collections c
LEFT JOIN customers cust
    ON c.customerid = cust.customer_id
WHERE c.status = 'Booked'
ORDER BY c.collectionid DESC;
