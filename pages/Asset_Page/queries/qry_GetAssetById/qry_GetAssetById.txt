SELECT
  a.*,
  it.item_type_name,
  mk.make_name,
  md.model_name,
  md.weight AS model_weight,
  cu.customer_name
FROM assets a
LEFT JOIN item_types it ON it.item_type_id = a.item_type::integer
LEFT JOIN makes mk ON mk.make_id = a.make::integer
LEFT JOIN models md ON md.model_id = a.model::integer
LEFT JOIN collections c ON c.collection_id = a.collection_id
LEFT JOIN customers cu ON cu.customer_id = c.customer_id
WHERE a.asset_id = {{appsmith.URL.queryParams.asset_id}};
