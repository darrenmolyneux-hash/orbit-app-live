SELECT 
  hp.id,
  hp.part_ref,
  pt.name AS part_type,
  hp.spec,
  hp.serial_number,
  hp.condition_grade,
  hp.status,
  hp.removed_by,
  hp.removed_at::date AS removed_date,
  hp.disposition_due,
  hp.removal_notes,
  hp.installed_in_asset_id,
  a.asset_ref AS installed_in_ref,
  CASE 
    WHEN hp.disposition_due < CURRENT_DATE THEN true 
    ELSE false 
  END AS is_overdue
FROM harvested_parts hp
LEFT JOIN part_types pt ON pt.part_type_id = hp.part_type_id
LEFT JOIN assets a ON a.asset_id = hp.installed_in_asset_id
WHERE hp.source_asset_id = {{appsmith.URL.queryParams.asset_id}}
ORDER BY hp.removed_at DESC;
