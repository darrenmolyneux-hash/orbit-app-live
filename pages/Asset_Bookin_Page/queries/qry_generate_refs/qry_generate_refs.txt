CREATE OR REPLACE FUNCTION generate_asset_refs(p_collection_ref INT, p_qty INT)
RETURNS TABLE(asset_ref TEXT) AS $$
DECLARE
    current_ref INT;
BEGIN
    -- Validate input
    IF p_collection_ref IS NULL THEN
        RAISE EXCEPTION 'Collection ref cannot be NULL';
    END IF;

    IF p_qty IS NULL OR p_qty <= 0 THEN
        RAISE EXCEPTION 'Quantity must be greater than 0';
    END IF;

    -- Ensure the collection exists in the counter table
    INSERT INTO collection_asset_refs (collection_id, next_ref)
    VALUES (p_collection_ref, 1)
    ON CONFLICT (collection_id) DO NOTHING;

    -- Lock the row and get the current next_ref
    SELECT next_ref INTO current_ref
    FROM collection_asset_refs
    WHERE collection_id = p_collection_ref
    FOR UPDATE;

    -- Increment the counter by the quantity requested
    UPDATE collection_asset_refs
    SET next_ref = next_ref + p_qty
    WHERE collection_id = p_collection_ref;

    -- Return p_qty asset refs
    RETURN QUERY
    SELECT p_collection_ref || '-' || LPAD((current_ref + s)::text, 4, '0')
    FROM generate_series(0, p_qty - 1) AS s;
END;
$$ LANGUAGE plpgsql;
